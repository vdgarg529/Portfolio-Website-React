

# Create window

```bash
ctrl + b c
```

# Rename current window

```bash
ctrl + b ,
```

# Close/Kill current window

```bash
ctrl + b &
```

# List windows

```bash
ctrl + b w
```

# Previous window

```bash
ctrl + b p
```

# Next window

```bash
ctrl + b n
```

# Switch/select window by number

```bash
ctrl + b 0....9
```

# Toggle last active window

```bash
ctrl + b l
```


# Reorder window, swap window number 2(src) and 1(dst)

```bash
swap-window -s 2 -t 1
```

# Move current window to the left by one position

```bash
swap-window -t -1
```

# Move window from source to target

```bash
move-window -s src_ses:win -t target_ses:win
#or
movew -s foo:0 -t bar:9
#or
movew -s 0:0 -t 1:9
```

# Reposition window in the current session

```bash
move-window -s src_session:src_window
#or
movew -s 0:9
```

# Renumber windows to remove gap in the sequence

```bash
move-window -r
#or
movew -r
```
